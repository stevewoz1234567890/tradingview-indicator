// Marker Detection (Lookback, bounded)

f_markerBullLookback() =>
    // Straight: Bull -> Bear(s) -> Bull within lookback
    bool straightOK = false
    if useStraightMarker
        for L = 2 to markerLookback
            if close[L] > open[L]  // lead bull
                for T = 1 to L - 1
                    if close[T] > open[T]  // tail bull
                        bool allBear = true
                        for m = T + 1 to L - 1
                            if close[m] >= open[m]
                                allBear := false
                        if allBear
                            straightOK := true
                            break
                if straightOK
                    break

    // Swing: 3-bar V shape inside lookback (bullish V)
    bool swingOK = false
    if useSwingMarker
        for m = 2 to markerLookback - 1
            leftH = high[m + 1]
            midH  = high[m]
            rightH= high[m - 1]
            leftL = low[m + 1]
            midL  = low[m]
            rightL= low[m - 1]
            if (leftH > midH) and (rightH > midH) and (leftL > midL) and (rightL > midL)
                swingOK := true
                break

    requireBothMarkers ? (straightOK and swingOK) : (straightOK or swingOK or (not useStraightMarker and not useSwingMarker))

f_markerBearLookback() =>
    // Straight: Bear -> Bull(s) -> Bear
    bool straightOK = false
    if useStraightMarker
        for L = 2 to markerLookback
            if close[L] < open[L]  // lead bear
                for T = 1 to L - 1
                    if close[T] < open[T]  // tail bear
                        bool allBull = true
                        for m = T + 1 to L - 1
                            if close[m] <= open[m]
                                allBull := false
                        if allBull
                            straightOK := true
                            break
                if straightOK
                    break

    // Swing: inverted V (bearish)
    bool swingOK = false
    if useSwingMarker
        for m = 2 to markerLookback - 1
            leftL = low[m + 1]
            midL  = low[m]
            rightL= low[m - 1]
            leftH = high[m + 1]
            midH  = high[m]
            rightH= high[m - 1]
            if (leftL < midL) and (rightL < midL) and (leftH < midH) and (rightH < midH)
                swingOK := true
                break

    requireBothMarkers ? (straightOK and swingOK) : (straightOK or swingOK or (not useStraightMarker and not useSwingMarker))

f_cleanStraightBull() =>

// Multi clean: entire segment after opposite bar must be impulse-only; straight = local high/low peak on 3 bars (mo = offset of straight candle from current bar).
f_allImpulseBarsSince(dir, lastOppBarIdx) =>
    bool ok = true
    if na(lastOppBarIdx) or lastOppBarIdx >= bar_index
        ok := false
    else
        for bi = lastOppBarIdx + 1 to bar_index
            off = bar_index - bi
            if off < 0 or off > 10000
                ok := false
                break
            if dir == 1 and not isBull[off]
                ok := false
                break
            if dir == -1 and close[off] > open[off]
                ok := false
                break
    ok

// Multi clean straight: middle bar's high (bull) / low (bear) strictly higher/lower than immediate neighbors; all three bars strictly impulse (no doji) — matches "tallest by high" 3-candle pattern, no bearish in pattern.
// Picks earliest straight in time (largest mo) when multiple qualify.
f_findMultiStraightOffset(dir, lastOppBar) =>
    int result = int(na)
    if not na(lastOppBar) and lastOppBar < bar_index - 1 and f_allImpulseBarsSince(dir, lastOppBar)
        int maxMo = math.min(markerLookback, bar_index - lastOppBar - 1)
        maxMo := math.min(maxMo, bar_index - 1)
        if maxMo >= 1
            for j = 0 to maxMo - 1
                int mo = maxMo - j
                if bar_index - mo <= lastOppBar
                    continue
                bool pat = false
                if dir == 1
                    pat := high[mo] > high[mo + 1] and high[mo] > high[mo - 1] and isStrictBull[mo] and isStrictBull[mo + 1] and isStrictBull[mo - 1]
                else
                    pat := low[mo] < low[mo + 1] and low[mo] < low[mo - 1] and isStrictBear[mo] and isStrictBear[mo + 1] and isStrictBear[mo - 1]
                if pat
                    result := mo
                    break
    result

//────────────────────────────────────────
// Partial Break checks (bounded, publish-safe)
// PB Bull: after confirm, find bear group then bull breaks above bear high
// PB Bear: after confirm, find bull group then bear breaks below bull low
//────────────────────────────────────────
f_pbBull(confirmBarIndex) =>
    bool pb = false
    float highestBearHigh = na
    bool foundBearGroup = false
    int age = bar_index - confirmBarIndex
    if age > 0
        for x = 1 to pbLookback
            i = pbLookback -x + 1
            if i <= age
                if close[i] < open[i]
                    foundBearGroup := true
                    highestBearHigh := na(highestBearHigh) ? high[i] : math.max(highestBearHigh, high[i])
                else if close[i] > open[i] and foundBearGroup and not na(highestBearHigh)
                    if high > highestBearHigh or high[i] > highestBearHigh
                        pb := true
                        break
    pb

f_pbBear(confirmBarIndex) =>
    bool pb = false
    float lowestBullLow = na
    bool foundBullGroup = false
    int age = bar_index - confirmBarIndex
    if age > 0
        for x = 1 to pbLookback
            i = pbLookback - x + 1
            if i <= age
                if close[i] > open[i]
                    foundBullGroup := true
                    lowestBullLow := na(lowestBullLow) ? low[i] : math.min(lowestBullLow, low[i])
                else if close[i] < open[i] and foundBullGroup and not na(lowestBullLow)
                    if low < lowestBullLow or low[i] < lowestBullLow
                        pb := true
                        break
    pb

f_applyWickRule(_top, _bot, _open, _nextHigh, _nextLow, _dir) =>
    float zTop = _top
    float zBot = _bot

    if useWickRule
        if _dir == 1 and _top > _nextHigh
            zTop := _open
        if _dir == -1 and _bot < _nextLow
            zBot := _open

    [zTop, zBot]

f_findLastOppositeBar(dir, maxLookback) =>
    int foundBar = na
    for i = 1 to maxLookback
        if dir == 1 and close[i] < open[i]  // bearish for bullish OB
            foundBar := bar_index[i]
            break
        if dir == -1 and close[i] >= open[i] // bullish for bearish OB
            foundBar := bar_index[i]
            break
    foundBar

f_sameDirTouch(_dir, _top, _bot, obLeft, _isCleanMarkerWait) =>    
    // same-direction candle
    sameDir = (_dir == 1 and isBear) or (_dir == -1 and isBull)

    // fully inside the zone
    contained = high <= _top and low >= _bot

    // boundary touch (one-sided)
    bullTouch = _dir == 1 and ((low <= _top and high > _top) or (low < _bot and high <= _top))
    bearTouch = _dir == -1 and ((high >= _bot and low < _bot) or (high >= _top and low >= _bot))

    // full override (must be excluded)
    override = high > _top and low < _bot

    // jump zone
    bullJump = _dir == 1 and low < _bot and high < _bot
    bearJump = _dir == -1 and low > _top and high > _top
    
    if sameDir
        contained or bullTouch or bearTouch or override or bullJump or bearJump
    else if obLeft != -1 and obLeft < bar_index - 2
        if _isCleanMarkerWait
            contained or bullTouch or bearTouch or override or bullJump or bearJump
        else
            int oppCount = 0
            for x = obLeft + 2 to bar_index - 1
                if bar_index - x >= 0 and bar_index - x <= 100
                    if _dir == 1 ? close[bar_index - x] < open[bar_index - x] : close[bar_index - x] > open[bar_index - x]// and not (_dir == 1 ? isBear[bar_index - x + 1] : isBull[bar_index - x - 1])
                        oppCount += 1
            if oppCount > 1
                contained or bullTouch or bearTouch or override or bullJump or bearJump

//────────────────────────────────────────
// Build OB zone with Wick Rule option
// Bullish OB comes from last bearish candle before bull break
// Bearish OB comes from last bullish candle before bear break
//────────────────────────────────────────
f_calcBullZone(refOffset) =>
    float top = high[refOffset]
    float bot = low[refOffset]
    if useWickRule
        // If ref wick extends above the next candle high, cap at ref open (body-top)
        nextH = refOffset > 0 ? high[refOffset - 1] : na
        refO  = open[refOffset]
        if not na(nextH) and high[refOffset] > nextH
            top := refO
    [top, bot]

f_calcBearZone(refOffset) =>
    float top = high[refOffset]
    float bot = low[refOffset]
    if useWickRule
        // If ref wick extends below the next candle low, raise bottom to ref open (body-bottom)
        nextL = refOffset > 0 ? low[refOffset - 1] : na
        refO  = open[refOffset]
        if not na(nextL) and low[refOffset] < nextL
            bot := refO
    [top, bot]

    else

        for i = 0 to array.size(obBoxes) - 1
            bx = array.get(obBoxes, i)
            if box.get_left(bx) == _left
                exists := true
                break
        exists

// HTF candle range vs stored zones (for touch alerts)
f_zonesTouchedInRange(_tops, _bots, _dirs, rh, rl) =>
    bool hitBull = false
    bool hitBear = false
    int zn = array.size(_tops)
    if zn > 0 and not na(rh) and not na(rl)
        for zi = 0 to zn - 1
            float zt = array.get(_tops, zi)
            float zb = array.get(_bots, zi)
            int zd = array.get(_dirs, zi)
            bool ov = math.max(rh, rl) >= zb and math.min(rh, rl) <= zt
            if ov
                hitBull := hitBull or zd == 1
                hitBear := hitBear or zd == -1
    [hitBull, hitBear]

// Updated OB and Clean Marker Logic with distinct OB start
for d = 0 to 1
    dir = d == 0 ? 1 : -1

    startContext = (dir == 1 and bullBreak) or (dir == -1 and bearBreak)

    if startContext and array.get(cmState, d) == CleanState.IDLE
        array.set(cmState, d, CleanState.SCANNING)

    if array.get(cmState, d) == CleanState.SCANNING
        // Clean marker: strict 3-bar straight (high/low vs neighbors); from opposite bar to here, no bearish (bull) / no bullish (bear) — same as multi-clean segment rule
        cleanStraight = (dir == 1 ? isStrictBull[2] and isStrictBull[1] and isStrictBull[0] : isStrictBear[2] and isStrictBear[1] and isStrictBear[0]) and (dir == 1 ? high[1] > high[2] and high[1] > high[0] : low[1] < low[2] and low[1] < low[0])
        cleanAllBullBearOk = not na(lastOppBar) and f_allImpulseBarsSince(dir, lastOppBar)

        if f_isOpposite(dir)
            array.set(cmState, d, CleanState.IDLE)
        
        else if bar_index >= 2 and cleanStraight and cleanAllBullBearOk
            // Store straight marker reference for CLEAN break
            if dir == 1
                array.set(cmBreakRef, d, high[1])   // straight bullish candle high
                array.set(cmImpulse, d, low[1])     // start lowest tracking
            else
                array.set(cmBreakRef, d, low[1])    // straight bearish candle low
                array.set(cmImpulse, d, high[1])    // start highest tracking
            
            if not na(lastOppBar)
                offset = bar_index - lastOppBar

                refHigh = high[offset]
                refLow  = low[offset]
                refOpen = open[offset]

                // Apply General Marker Wick Rule immediately
                [adjTop, adjBot] = f_applyWickRule(refHigh, refLow, refOpen, high[offset - 1], low[offset - 1], dir)

                if dir == 1
                    array.set(cmExtreme, d, adjTop)   // bullish OB top
                    array.set(cmZoneLow, d, adjBot)   // bullish OB bottom
                else
                    array.set(cmExtreme, d, adjBot)   // bearish OB bottom
                    array.set(cmZoneHigh, d, adjTop)  // bearish OB top

                array.set(cmLeft, d, lastOppBar)
                array.set(cmState, d, CleanState.WAIT_TAKE)
               
    // WAIT_TAKE phase
    if array.get(cmState, d) == CleanState.WAIT_TAKE
        if f_isOpposite(dir)
            array.set(cmState, d, CleanState.IDLE)
            array.set(cmExtremeTaken, d, false)
            array.set(cmBreakRef, d, na)
            array.set(cmImpulse, d, na)
        else
            zoneHigh = dir == 1 ? array.get(cmExtreme, d) : array.get(cmZoneHigh, d)
            zoneLow  = dir == 1 ? array.get(cmZoneLow, d) : array.get(cmExtreme, d)

            if f_sameDirTouch(dir, zoneHigh, zoneLow, array.get(cmLeft, d), true)
                array.set(cmState, d, CleanState.IDLE)
                array.set(cmExtremeTaken, d, false)
                array.set(cmBreakRef, d, na)
                array.set(cmImpulse, d, na)

            else
                // update impulse extreme
                imp = array.get(cmImpulse, d)

                if dir == 1
                    imp := math.min(imp, low)
                else
                    imp := math.max(imp, high)

                array.set(cmImpulse, d, imp)

                // Arm the OB when extreme is taken
                if not array.get(cmExtremeTaken, d) and f_extremeTaken(dir, array.get(cmExtreme, d))
                    array.set(cmExtremeTaken, d, true)

                cleanBreak = barstate.isconfirmed and (dir == 1 ? close > array.get(cmBreakRef, d) : close < array.get(cmBreakRef, d))

                    rawTop = dir == 1 ? array.get(cmExtreme, d) : array.get(cmZoneHigh, d)
                    rawBot = dir == 1 ? array.get(cmZoneLow, d) : array.get(cmExtreme, d)

                    [zoneHigh, zoneLow] = f_applyWickRule(rawTop, rawBot, open, high, low, dir)

                    if f_distanceOk(dir == 1 ? zoneHigh : zoneLow, impulseExtreme) and useCleanMarker and not f_boxLeftExists(array.get(cmLeft, d))
                        if showMarkerHighlight

                        bx = box.new( array.get(cmLeft, d), zoneHigh, bar_index, zoneLow, bgcolor = dir == 1 ? bullFill : bearFill, border_color = dir == 1 ? bullBorder : bearBorder )
                        array.push(obBoxes, bx)
                        array.push(obTops, zoneHigh)
                        array.push(obBots, zoneLow)
                        array.push(obDirs, dir)

                        if dir == 1
                            bullConfirmed := true
                        else
                            bearConfirmed := true

                    // reset
                    array.set(cmState, d, CleanState.IDLE)
                    array.set(cmExtremeTaken, d, false)
                    array.set(cmBreakRef, d, na)
                    array.set(cmImpulse, d, na)

// Multi Clean Marker (parallel pipeline; wider straight-candle detection per spec)
for d = 0 to 1
    dir = d == 0 ? 1 : -1

    startContextM = (dir == 1 and bullBreak) or (dir == -1 and bearBreak)

    if startContextM and array.get(mcmState, d) == CleanState.IDLE
        array.set(mcmState, d, CleanState.SCANNING)

    lastOppBarM = f_findLastOppositeBar(dir, markerLookback)

    if array.get(mcmState, d) == CleanState.SCANNING
        if f_isOpposite(dir)
            array.set(mcmState, d, CleanState.IDLE)
        else
            int mo = f_findMultiStraightOffset(dir, lastOppBarM)
            if not na(mo) and not na(lastOppBarM)
                if dir == 1
                    array.set(mcmBreakRef, d, high[mo])
                    array.set(mcmImpulse, d, low[mo])
                else
                    array.set(mcmBreakRef, d, low[mo])
                    array.set(mcmImpulse, d, high[mo])

                offsetM = bar_index - lastOppBarM

                refHighM = high[offsetM]
                refLowM  = low[offsetM]
                refOpenM = open[offsetM]

                [adjTopM, adjBotM] = f_applyWickRule(refHighM, refLowM, refOpenM, high[offsetM - 1], low[offsetM - 1], dir)

                if dir == 1
                    array.set(mcmExtreme, d, adjTopM)
                    array.set(mcmZoneLow, d, adjBotM)
                else
                    array.set(mcmExtreme, d, adjBotM)
                    array.set(mcmZoneHigh, d, adjTopM)

                array.set(mcmLeft, d, lastOppBarM)
                array.set(mcmState, d, CleanState.WAIT_TAKE)

    if array.get(mcmState, d) == CleanState.WAIT_TAKE
        if f_isOpposite(dir)
            array.set(mcmState, d, CleanState.IDLE)
            array.set(mcmExtremeTaken, d, false)
            array.set(mcmBreakRef, d, na)
            array.set(mcmImpulse, d, na)
        else
            zoneHighM = dir == 1 ? array.get(mcmExtreme, d) : array.get(mcmZoneHigh, d)
            zoneLowM  = dir == 1 ? array.get(mcmZoneLow, d) : array.get(mcmExtreme, d)

            if f_sameDirTouch(dir, zoneHighM, zoneLowM, array.get(mcmLeft, d), true)
                array.set(mcmState, d, CleanState.IDLE)
                array.set(mcmExtremeTaken, d, false)
                array.set(mcmBreakRef, d, na)
                array.set(mcmImpulse, d, na)

            else
                impM = array.get(mcmImpulse, d)

                if dir == 1
                    impM := math.min(impM, low)
                else
                    impM := math.max(impM, high)

                array.set(mcmImpulse, d, impM)

                if not array.get(mcmExtremeTaken, d) and f_extremeTaken(dir, array.get(mcmExtreme, d))
                    array.set(mcmExtremeTaken, d, true)

                // Straight level must be taken by a *closed* candle (close through ref), not wick-only
                mcleanBreak = barstate.isconfirmed and (dir == 1 ? close > array.get(mcmBreakRef, d) : close < array.get(mcmBreakRef, d))

                if array.get(mcmExtremeTaken, d) and mcleanBreak
                    rawTopM = dir == 1 ? array.get(mcmExtreme, d) : array.get(mcmZoneHigh, d)
                    rawBotM = dir == 1 ? array.get(mcmZoneLow, d) : array.get(mcmExtreme, d)

                    [zoneHighM2, zoneLowM2] = f_applyWickRule(rawTopM, rawBotM, open, high, low, dir)

                    impulseExtremeM = array.get(mcmImpulse, d)

                    if f_distanceOk(dir == 1 ? zoneHighM2 : zoneLowM2, impulseExtremeM) and useMultiCleanMarker and not f_boxLeftExists(array.get(mcmLeft, d))
                        if showMarkerHighlight
                            label.new(bar_index - 1, dir == 1 ? zoneHighM2 : zoneLowM2, "MClean", style = dir == 1 ? label.style_label_down : label.style_label_up, color = dir == 1 ? color.new(color.teal, 77) : color.new(color.orange, 77), textcolor = color.white, size = size.normal)
                        bxM = box.new(array.get(mcmLeft, d), zoneHighM2, bar_index, zoneLowM2, bgcolor = dir == 1 ? bullFill : bearFill, border_color = dir == 1 ? bullBorder : bearBorder)
                        array.push(obBoxes, bxM)
                        array.push(obTops, zoneHighM2)
                        array.push(obBots, zoneLowM2)
                        array.push(obDirs, dir)

                        if dir == 1
                            bullConfirmed := true
                        else
                            bearConfirmed := true

                    array.set(mcmState, d, CleanState.IDLE)
                    array.set(mcmExtremeTaken, d, false)
                    array.set(mcmBreakRef, d, na)
                    array.set(mcmImpulse, d, na)

f_bmHighLow(_dir, maxLookback) =>
    bool isOpp = false
    float _bmHigh = na
    float _bmLow = na
    float _bmPreHigh = na
    float _bmPreLow = na
    for i = 1 to maxLookback
        if _dir == 1 ? close[i] < open[i] : close[i] > open[i]
            isOpp := true
            _bmHigh := na(_bmHigh) ? high[i] : math.max(_bmHigh, high[i])
            _bmLow := na(_bmLow) ? low[i] : math.min(_bmLow, low[i])
        else if isOpp == true
            _bmPreHigh := high[i]
            _bmPreLow := low[i]
            break
    [_bmHigh, _bmLow, _bmPreHigh, _bmPreLow]

//────────────────────
// BREAK MARKER Logic (Fixed Slot Allocation)
//────────────────────
for d = 0 to 1
    dir = d == 0 ? 1 : -1
    base = d * maxBMSeq

    bos = f_isBreakContext(dir)

    //────────────────────
    // BOS → allocate FREE slot
    //────────────────────
    if bos
        int freeSlot = na

        for s = 0 to maxBMSeq - 1
            idx = base + s
            if array.get(bmState, idx) == BreakState.IDLE
                freeSlot := idx
                break

        if not na(freeSlot)
            array.set(bmLocked, freeSlot, false)

            array.set(bmState, freeSlot, BreakState.COLLECT_OPPOSITE)

            array.set(bmLocked, freeSlot, false)
            array.set(bmHigh, freeSlot, na)
            array.set(bmLow, freeSlot, na)
            array.set(bmPreHigh, freeSlot, na)
            array.set(bmPreLow, freeSlot, na)

            array.set(bmLeft, freeSlot, na)
            array.set(bmOBHigh, freeSlot, na)
            array.set(bmOBLow, freeSlot, na)
    
    lastOppBar = f_findLastOppositeBar(dir, markerLookback)

    //────────────────────
    // PROCESS ACTIVE SLOTS
    //────────────────────
    for s = 0 to maxBMSeq - 1
        i = base + s
        state = array.get(bmState, i)

        if state == BreakState.IDLE
            continue
        
        left = array.get(bmLeft, i)

        if not na(left)
            if bar_index - left > markerLookback
                array.set(bmState, i, BreakState.IDLE)
                array.set(bmLeft, i, na)
                array.set(bmOBHigh, i, na)
                array.set(bmOBLow, i, na)
                array.set(bmHigh, i, na)
                array.set(bmLow, i, na)
                array.set(bmPreHigh, i, na)
                array.set(bmPreLow, i, na)
                array.set(bmLocked, i, false)
                continue

        //────────────────────
        // COLLECT OPPOSITE
        //────────────────────
        if state == BreakState.COLLECT_OPPOSITE

            if f_isOpposite(dir)
                h = array.get(bmHigh, i)
                l = array.get(bmLow, i)


                array.set(bmHigh, i, na(h) ? high : math.max(h, high))
                array.set(bmLow, i, na(l) ? low : math.min(l, low))
                if not na(lastOppBar)

                    array.set(bmLeft, i, lastOppBar)

                    offset = bar_index - lastOppBar

                    refHigh = high[offset]
                    refLow  = low[offset]
                    refOpen = open[offset]

                    [adjTop, adjBot] = f_applyWickRule(refHigh, refLow, refOpen, high[offset - 1], low[offset - 1], dir)

                    array.set(bmOBHigh, i, adjTop)
                    array.set(bmOBLow,  i, adjBot)

                    array.set(bmState, i, BreakState.WAIT_BREAK)

        //────────────────────
        // WAIT BREAK
        //────────────────────
        if array.get(bmState, i) == BreakState.WAIT_BREAK and barstate.isconfirmed

            // cancel if touched
            if f_sameDirTouch(dir,array.get(bmOBHigh, i), array.get(bmOBLow, i),array.get(bmLeft, i), false)

                array.set(bmState, i, BreakState.IDLE)
                array.set(bmHigh, i, na)
                array.set(bmLow, i, na)
                array.set(bmLeft, i, na)
                array.set(bmOBHigh, i, na)
                array.set(bmOBLow, i, na)

            // impulse break
            else if f_isImpulse(dir)

                [fbmHigh, fbmLow, fbmPreHigh, fbmPreLow] = f_bmHighLow(dir, markerLookback)

                array.set(bmHigh, i, fbmHigh)
                array.set(bmLow, i, fbmLow)

                // if dir == -1
                array.set(bmPreLow, i, fbmPreLow)
                array.set(bmPreHigh, i, fbmPreHigh)

                highestBearish = array.get(bmHigh, i)
                lowestBullish  = array.get(bmLow, i)

                // if dir == 1
                highestBearish := math.max(highestBearish, array.get(bmPreHigh, i))
                lowestBullish := math.min(lowestBullish, array.get(bmPreLow, i))

                refBreak = dir == 1 ? highestBearish : lowestBullish
                broken = dir == 1 ? high > refBreak : low < refBreak

                if broken and not array.get(bmLocked, i)

                    zoneHigh = array.get(bmOBHigh, i)
                    zoneLow  = array.get(bmOBLow, i)

                    if f_distanceOk(dir == 1 ? zoneHigh : zoneLow, dir == 1 ? lowestBullish : highestBearish) and useBreakMarker and not f_boxLeftExists(array.get(bmLeft, i))
                        if showMarkerHighlight
                            label.new(bar_index - 1, dir == 1 ? zoneHigh : zoneLow, "Break", style = dir == 1 ? label.style_label_down : label.style_label_up, color = dir == 1 ? color.new(color.blue, 77) : color.new(color.red, 77), textcolor = color.white, size = size.normal)

                        bx = box.new( array.get(bmLeft, i), zoneHigh, bar_index, zoneLow, bgcolor = dir == 1 ? bullFill : bearFill, border_color = dir == 1 ? bullBorder : bearBorder)

                        array.push(obBoxes, bx)
                        array.push(obTops, zoneHigh)
                        array.push(obBots, zoneLow)
                        array.push(obDirs, dir)

                        array.set(bmLocked, i, true)

                        if dir == 1
                            bullConfirmed := true
                        else
                            bearConfirmed := true

                        // reset slot
                        array.set(bmState, i, BreakState.IDLE)
                        array.set(bmLeft, i, na)
                        array.set(bmOBHigh, i, na)
                        array.set(bmOBLow, i, na)
