// Other forex pairs: pip = 0.0001 (4th decimal place)

    pipSize := 0.1

    pipSize := syminfo.mintick

// GENERAL MARKER LOOKBACK — f_markerBull/BearLookback: straight chain + optional swing V inside markerLookback
//────────────────────────────────────────
f_markerBullLookback() =>
    // Straight: Bull -> Bear(s) -> Bull within lookback
    bool straightOK = false
    if useStraightMarker
        for L = 2 to markerLookback
            if close[L] > open[L]  // lead bull
                for T = 1 to L - 1
                    if close[T] > open[T]  // tail bull
                        bool allBear = true
                        for m = T + 1 to L - 1
                            if close[m] >= open[m]
                                allBear := false
                        if allBear
                            straightOK := true
                            break
                if straightOK
                    break

    // Swing: 3-bar V shape inside lookback (bullish V)
    bool swingOK = false
    if useSwingMarker
        for m = 2 to markerLookback - 1
            leftH = high[m + 1]
            midH  = high[m]
            rightH= high[m - 1]
            leftL = low[m + 1]
            midL  = low[m]
            rightL= low[m - 1]
            if (leftH > midH) and (rightH > midH) and (leftL > midL) and (rightL > midL)
                swingOK := true
                break

    requireBothMarkers ? (straightOK and swingOK) : (straightOK or swingOK or (not useStraightMarker and not useSwingMarker))

f_markerBearLookback() =>
    // Straight: Bear -> Bull(s) -> Bear
    bool straightOK = false
    if useStraightMarker
        for L = 2 to markerLookback
            if close[L] < open[L]  // lead bear
                for T = 1 to L - 1
                    if close[T] < open[T]  // tail bear
                        bool allBull = true
                        for m = T + 1 to L - 1
                            if close[m] <= open[m]
                                allBull := false
                        if allBull
                            straightOK := true
                            break
                if straightOK
                    break

    // Swing: inverted V (bearish)
    bool swingOK = false
    if useSwingMarker
        for m = 2 to markerLookback - 1
            leftL = low[m + 1]
            midL  = low[m]
            rightL= low[m - 1]
            leftH = high[m + 1]
            midH  = high[m]
            rightH= high[m - 1]
            if (leftL < midL) and (rightL < midL) and (leftH < midH) and (rightH < midH)
                swingOK := true
                break

    requireBothMarkers ? (straightOK and swingOK) : (straightOK or swingOK or (not useStraightMarker and not useSwingMarker))

//────────────────────────────────────────
// CLEAN / MULTI-CLEAN HELPERS — impulse-only segment, opposite before pivot, straight pivot search, impulse fold
//────────────────────────────────────────
// Multi clean: entire segment after opposite bar must be impulse-only when requireImpulseAfterOpp; straight = local high/low peak on 3 bars (mo = offset of straight candle from current bar).
// Clean marker: same 3-bar wick pivot but all three candles must be impulse (bull or bear); segment after last opposite before pivot must be impulse-only (no bear in a bull leg — else Break marker only).
f_allImpulseBarsSince(dir, lastOppBarIdx) =>
    bool ok = true
    if na(lastOppBarIdx) or lastOppBarIdx >= bar_index
        ok := false
    else
        for bi = lastOppBarIdx + 1 to bar_index
            off = bar_index - bi
            if off < 0 or off > bar_index
                ok := false
                break
            if dir == 1 and not isBull[off]
                ok := false
                break
            if dir == -1 and close[off] > open[off]
                ok := false
                break
    ok

// Nearest opposite-direction bar *older* than straightBarIdx (for zone left). Bull → bear; bear → bull.
f_lastOppBarBefore(dir, straightBarIdx) =>
    int res = int(na)
    if not na(straightBarIdx) and straightBarIdx > 0
        for step = 1 to markerLookback
            int bi = straightBarIdx - step
            if bi < 0
                break
            int off = bar_index - bi
            if off < 0 or off > bar_index
                break
            bool oppB = dir == 1 ? isBear[off] : isBull[off]
            if oppB
                res := bi
                break
    res

// Clean marker: nearest3-bar straight wick pivot where all three bars are impulse (strict bull / strict bear); segment after last opposite before pivot is impulse-only; INV on first bar after “candle 3”.
f_findCleanStraightOffset(dir) =>
    int result = int(na)
    int maxMo = math.min(markerLookback, bar_index - 1)
    if maxMo >= 1
        for mo = 1 to maxMo
            bool pat = false
            if dir == 1
                pat := high[mo] > high[mo + 1] and high[mo] > high[mo - 1]
                pat := pat and close[mo] > open[mo] and close[mo + 1] > open[mo + 1] and close[mo - 1] > open[mo - 1]
            else
                pat := low[mo] < low[mo + 1] and low[mo] < low[mo - 1]
                pat := pat and close[mo] < open[mo] and close[mo + 1] < open[mo + 1] and close[mo - 1] < open[mo - 1]
            if pat
                int straightBar = bar_index - mo
                int lastOpp = f_lastOppBarBefore(dir, straightBar)
                bool segAllImpulse = not na(lastOpp) and f_allImpulseBarsSince(dir, lastOpp)
                if segAllImpulse and f_pivotAfterC3Ok(dir, mo)
                    result := mo
                    break
    result

// Multi clean: 3-bar pivot on highs/lows — middle candle (“candle 2”) can be any color; only wick extreme vs neighbors matters. Nearest match (smallest mo) breaks first.
f_findMultiStraightOffset(dir) =>
    int result = int(na)
    int maxMo = math.min(markerLookback, bar_index - 1)
    if maxMo >= 1
        for mo = 1 to maxMo
            bool pat = false
            if dir == 1
                pat := high[mo] > high[mo + 1] and high[mo] > high[mo - 1]
            else
                pat := low[mo] < low[mo + 1] and low[mo] < low[mo - 1]
            if pat
                result := mo
                break
    result

// Lowest low from straight candle (offset strOff) through current bar (incl. bars between straight and now).
f_minLowStraightToNow(strOff) =>
    float m = low[strOff]
    if strOff >= 1
        for k = 0 to strOff - 1
            m := math.min(m, low[k])
    m

// Highest high from straight candle through current bar (bear multi clean).
f_maxHighStraightToNow(strOff) =>
    float m = high[strOff]
    if strOff >= 1
        for k = 0 to strOff - 1
            m := math.max(m, high[k])
    m

// Candle 3 is at [mo-1]; first bar to its right is [mo-2]. If that bar is opposite color (bear after bull fractal / bull after bear fractal), setup is invalid.
f_pivotAfterC3Ok(dir, mo) =>
    mo < 2 ? true : (dir == 1 ? not isBear[mo - 2] : not isBull[mo - 2])

// Multi clean only — can skip INV filter via input (Clean marker always uses f_pivotAfterC3Ok)
f_mcmPivotAfterOk(dir, mo) =>
    mcmSkipInvalidCandleRule ? true : f_pivotAfterC3Ok(dir, mo)

// First bar after pivot (chronological: off mo-1, mo-2, … 0) where close takes ref (same test as mcleanBreak close vs break ref).
f_mcmFirstConfirmOff(dir, mo, ref) =>
    int res = int(na)
    if mo >= 1
        for step = 0 to mo - 1
            int off = mo - 1 - step
            bool taken = dir == 1 ? close[off] > ref : close[off] < ref
            if taken
                res := off
                break
    res

// Multi clean: among pivots passing segment + pivotAfterC3, pick mo whose take ref is crossed earliest (min bar_index). If none crossed yet, fallback = nearest valid pivot (smallest mo).
f_findMultiStraightOffsetConfirmFirst(dir) =>
    int bestMo = int(na)
    int bestBar = int(na)
    int fallbackMo = int(na)
    int maxMo = math.min(markerLookback, bar_index - 1)
    if maxMo >= 1
        for mo = 1 to maxMo
            bool pat = false
            if dir == 1
                pat := high[mo] > high[mo + 1] and high[mo] > high[mo - 1]
            else
                pat := low[mo] < low[mo + 1] and low[mo] < low[mo - 1]
            if pat
                int straightBar = bar_index - mo
                int lastOppM = f_lastOppBarBefore(dir, straightBar)
                bool mcmSegmentOk = not na(lastOppM) and (not requireImpulseAfterOpp or f_allImpulseBarsSince(dir, lastOppM))
                if mcmSegmentOk and f_mcmPivotAfterOk(dir, mo)
                    if na(fallbackMo)
                        fallbackMo := mo
                    float ref = dir == 1 ? high[mo] : low[mo]
                    int fcOff = f_mcmFirstConfirmOff(dir, mo, ref)
                    if not na(fcOff)
                        int cBar = bar_index - fcOff
                        if na(bestBar) or cBar < bestBar or (cBar == bestBar and (na(bestMo) or mo < bestMo))
                            bestBar := cBar
                            bestMo := mo
    not na(bestMo) ? bestMo : fallbackMo

// PARTIAL BREAK (PB) — after sweep confirm bar, detect partial break via opposite cluster + reclaim
//────────────────────────────────────────
f_pbBull(confirmBarIndex) =>
    bool pb = false
    float highestBearHigh = na
    bool foundBearGroup = false
    int age = bar_index - confirmBarIndex
    if age > 0
        for x = 1 to pbLookback
            i = pbLookback -x + 1
            if i <= age
                if close[i] < open[i]
                    foundBearGroup := true
                    highestBearHigh := na(highestBearHigh) ? high[i] : math.max(highestBearHigh, high[i])
                else if close[i] > open[i] and foundBearGroup and not na(highestBearHigh)
                    if high > highestBearHigh or high[i] > highestBearHigh
                        pb := true
                        break
    pb

f_pbBear(confirmBarIndex) =>
    bool pb = false
    float lowestBullLow = na
    bool foundBullGroup = false
    int age = bar_index - confirmBarIndex
    if age > 0
        for x = 1 to pbLookback
            i = pbLookback - x + 1
            if i <= age
                if close[i] > open[i]
                    foundBullGroup := true
                    lowestBullLow := na(lowestBullLow) ? low[i] : math.min(lowestBullLow, low[i])
                else if close[i] < open[i] and foundBullGroup and not na(lowestBullLow)
                    if low < lowestBullLow or low[i] < lowestBullLow
                        pb := true
                        break
    pb

//────────────────────────────────────────
// WICK RULES — cap OB zone when wick clears next bar; pivot variant uses body edge
//────────────────────────────────────────
f_applyWickRule(_top, _bot, _open, _nextHigh, _nextLow, _dir) =>
    float zTop = _top
    float zBot = _bot

    if useWickRule
        if _dir == 1 and _top > _nextHigh
            zTop := _open
        if _dir == -1 and _bot < _nextLow
            zBot := _open

    [zTop, zBot]

// Pivot candle in 1–2–3 can be bullish or bearish; cap wick using true body edge (not raw open).
f_applyPivotWickRule(_top, _bot, _open, _close, _nextHigh, _nextLow, _dir) =>
    float bodyTop = math.max(_open, _close)
    float bodyBot = math.min(_open, _close)
    float zTop = _top
    float zBot = _bot
    if useWickRule
        if _dir == 1 and _top > _nextHigh
            zTop := bodyTop
        if _dir == -1 and _bot < _nextLow
            zBot := bodyBot
    [zTop, zBot]

// BOS (break of structure) — close through wick-adjusted last opposite extreme
//────────────────────────────────────────
[adjBearHigh, _] = f_applyWickRule(lastBearHigh, lastBearLow, lastBearOpen, high, low, 1)
[_, adjBullLow]  = f_applyWickRule(lastBullHigh, lastBullLow, lastBullOpen, high, low, -1)

bullBreak = not na(adjBearHigh) and close > adjBearHigh
bearBreak = not na(adjBullLow)  and close < adjBullLow

f_findLastOppositeBar(dir, maxLookback) =>
    int foundBar = na
    for i = 1 to maxLookback
        if dir == 1 and close[i] < open[i]  // bearish for bullish OB
            foundBar := bar_index[i]
            break
        if dir == -1 and close[i] >= open[i] // bullish for bearish OB
            foundBar := bar_index[i]
            break
    foundBar

f_sameDirTouch(_dir, _top, _bot, obLeft, _isCleanMarkerWait) =>    
    // same-direction candle
    sameDir = (_dir == 1 and isBear) or (_dir == -1 and isBull)

    // fully inside the zone
    contained = high <= _top and low >= _bot

    // boundary touch (one-sided)
    bullTouch = _dir == 1 and ((low <= _top and high > _top) or (low < _bot and high <= _top))
    bearTouch = _dir == -1 and ((high >= _bot and low < _bot) or (high >= _top and low >= _bot))

    // full override (must be excluded)
    override = high > _top and low < _bot

    // jump zone
    bullJump = _dir == 1 and low < _bot and high < _bot
    bearJump = _dir == -1 and low > _top and high > _top
    
    if sameDir
        contained or bullTouch or bearTouch or override or bullJump or bearJump
    else if obLeft != -1 and obLeft < bar_index - 2
        if _isCleanMarkerWait
            contained or bullTouch or bearTouch or override or bullJump or bearJump
        else
            int oppCount = 0
            for x = obLeft + 2 to bar_index - 1
                if bar_index - x >= 0 and bar_index - x <= 100
                    if _dir == 1 ? close[bar_index - x] < open[bar_index - x] : close[bar_index - x] > open[bar_index - x]// and not (_dir == 1 ? isBear[bar_index - x + 1] : isBull[bar_index - x - 1])
                        oppCount += 1
            if oppCount > 1
                contained or bullTouch or bearTouch or override or bullJump or bearJump

// ZONE BUILD + LOOKUPS — f_calcBull/BearZone, duplicate left check, HTF overlap for alerts

f_calcBullZone(refOffset) =>
    float top = high[refOffset]
    float bot = low[refOffset]
    if useWickRule
        // If ref wick extends above the next candle high, cap at ref open (body-top)
        nextH = refOffset > 0 ? high[refOffset - 1] : na
        refO  = open[refOffset]
        if not na(nextH) and high[refOffset] > nextH
            top := refO
    [top, bot]

f_calcBearZone(refOffset) =>
    float top = high[refOffset]
    float bot = low[refOffset]
    if useWickRule
        // If ref wick extends below the next candle low, raise bottom to ref open (body-bottom)
        nextL = refOffset > 0 ? low[refOffset - 1] : na
        refO  = open[refOffset]
        if not na(nextL) and low[refOffset] < nextL
            bot := refO
    [top, bot]

f_boxLeftExists(_left) =>
    if array.size(obBoxes) == 0
        false
    else
        bool exists = false
        for i = 0 to array.size(obBoxes) - 1
            bx = array.get(obBoxes, i)
            if box.get_left(bx) == _left
                exists := true
                break
        exists

// HTF candle range vs stored zones (for touch alerts)
f_zonesTouchedInRange(_tops, _bots, _dirs, rh, rl) =>
    bool hitBull = false
    bool hitBear = false
    int zn = array.size(_tops)
    if zn > 0 and not na(rh) and not na(rl)
        for zi = 0 to zn - 1
            float zt = array.get(_tops, zi)
            float zb = array.get(_bots, zi)
            int zd = array.get(_dirs, zi)
            bool ov = math.max(rh, rl) >= zb and math.min(rh, rl) <= zt
            if ov
                hitBull := hitBull or zd == 1
                hitBear := hitBear or zd == -1
    [hitBull, hitBear]

for d = 0 to 1

    if startContext and array.get(cmState, d) == CleanState.IDLE
        array.set(cmState, d, CleanState.SCANNING)

    if array.get(cmState, d) == CleanState.SCANNING

        straightBarCl = not na(moCl) ? bar_index - moCl : na
        lastOppCl = not na(straightBarCl) ? f_lastOppBarBefore(dir, straightBarCl) : na

        if f_isOpposite(dir)
            array.set(cmState, d, CleanState.IDLE)
            array.set(cmTouchAlertFired, d, false)
        
        else if not na(moCl) and not na(lastOppCl)
            if dir == 1
                array.set(cmBreakRef, d, high[moCl])

            else
                array.set(cmBreakRef, d, low[moCl])

            float nHc = moCl > 0 ? high[moCl - 1] : na
            float nLc = moCl > 0 ? low[moCl - 1] : na
            // Pivot wick rule: if zone candle high > next candle high, top = body top (not raw open only — same as Multi / f_calc* zones)
            [adjTopC, adjBotC] = f_applyPivotWickRule(high[moCl], low[moCl], open[moCl], close[moCl], nHc, nLc, dir)

            if dir == 1
                array.set(cmExtreme, d, adjTopC)
            else
                array.set(cmExtreme, d, adjBotC)

            array.set(cmLeft, d, lastOppCl)
            array.set(cmTouchAlertFired, d, false)
            array.set(cmState, d, CleanState.WAIT_TAKE)
               
    // WAIT_TAKE phase

        if f_isOpposite(dir)

            array.set(cmExtremeTaken, d, false)
            array.set(cmBreakRef, d, na)
            array.set(cmImpulse, d, na)
            array.set(cmTouchAlertFired, d, false)
        else
            // Impulse extreme first — zone bounds depend on running min/max since pivot
            imp = array.get(cmImpulse, d)

            if dir == 1
                imp := math.min(imp, low)
            else
                imp := math.max(imp, high)

            array.set(cmImpulse, d, imp)

            zoneHigh = dir == 1 ? array.get(cmExtreme, d) : array.get(cmImpulse, d)
            zoneLow  = dir == 1 ? array.get(cmImpulse, d) : array.get(cmExtreme, d)

                array.set(cmState, d, CleanState.IDLE)

                array.set(cmBreakRef, d, na)
                array.set(cmImpulse, d, na)
                array.set(cmTouchAlertFired, d, false)

            else
                brk = array.get(cmBreakRef, d)
                ext = array.get(cmExtreme, d)
                bool cmWickCapped = not na(brk) and not na(ext) and ((dir == 1 and ext < brk) or (dir == -1 and ext > brk))
                bool zoneTouch = math.max(high, low) >= zoneLow and math.min(high, low) <= zoneHigh
                if cmWickCapped and zoneTouch and barstate.isconfirmed and not array.get(cmTouchAlertFired, d)

                    if dir == 1
                        bullConfirmed := true
                    else

                // Straight level taken by *close* on a strict impulse bar (bullish / bearish), not wick-only
                cleanBreak = barstate.isconfirmed and (dir == 1 ? close > array.get(cmBreakRef, d) and isStrictBull : close < array.get(cmBreakRef, d) and isStrictBear)

                if cleanBreak
                    rawTop = dir == 1 ? array.get(cmExtreme, d) : array.get(cmImpulse, d)
                    rawBot = dir == 1 ? array.get(cmImpulse, d) : array.get(cmExtreme, d)

                    if f_distanceOk(dir == 1 ? boxHi : boxLo, impulseExtreme) and useCleanMarker and not f_boxLeftExists(array.get(cmLeft, d))
                        bx = box.new( array.get(cmLeft, d), boxHi, bar_index, boxLo, bgcolor = dir == 1 ? bullFill : bearFill, border_color = dir == 1 ? bullBorder : bearBorder )
                        f_boxRegionLabel(array.get(cmLeft, d), bar_index, boxHi, "Clean marker", dir == 1 ? color.new(color.blue, 78) : color.new(color.red, 78))
                        array.push(obBoxes, bx)
                        array.push(obTops, boxHi)
                        array.push(obBots, boxLo)
                        array.push(obDirs, dir)

                        if not cmWickCapped
                            if dir == 1
                                bullConfirmed := true
                            else
                                bearConfirmed := true

                    // reset
                    array.set(cmState, d, CleanState.IDLE)
                    array.set(cmExtremeTaken, d, false)
                    array.set(cmBreakRef, d, na)
                    array.set(cmImpulse, d, na)
                    array.set(cmTouchAlertFired, d, false)

//────────────────────────────────────────
// BREAK MARKER — f_bmHighLow: opposite cluster + first same-dir bar after (for break ref)
//────────────────────────────────────────
f_bmHighLow(_dir, maxLookback) =>
    bool isOpp = false
    float _bmHigh = na
    float _bmLow = na
    float _bmPreHigh = na
    float _bmPreLow = na
    for i = 1 to maxLookback
        if _dir == 1 ? close[i] < open[i] : close[i] > open[i]
            isOpp := true
            _bmHigh := na(_bmHigh) ? high[i] : math.max(_bmHigh, high[i])
            _bmLow := na(_bmLow) ? low[i] : math.min(_bmLow, low[i])
        else if isOpp == true
            _bmPreHigh := high[i]
            _bmPreLow := low[i]
            break
    [_bmHigh, _bmLow, _bmPreHigh, _bmPreLow]

//────────────────────────────────────────
// BREAK MARKER PIPELINE — BOS opens slot; COLLECT opposite → WAIT_BREAK → impulse break → box
//────────────────────────────────────────
for d = 0 to 1
    dir = d == 0 ? 1 : -1
    base = d * maxBMSeq

    bos = f_isBreakContext(dir)

    //────────────────────
    // BOS — grab next IDLE break-marker slot for this direction
    //────────────────────
    if bos
        int freeSlot = na

        for s = 0 to maxBMSeq - 1
            idx = base + s
            if array.get(bmState, idx) == BreakState.IDLE
                freeSlot := idx
                break

        if not na(freeSlot)
            array.set(bmLocked, freeSlot, false)

            array.set(bmState, freeSlot, BreakState.COLLECT_OPPOSITE)

            array.set(bmLocked, freeSlot, false)
            array.set(bmHigh, freeSlot, na)
            array.set(bmLow, freeSlot, na)
            array.set(bmPreHigh, freeSlot, na)
            array.set(bmPreLow, freeSlot, na)

            array.set(bmLeft, freeSlot, na)
            array.set(bmOBHigh, freeSlot, na)
            array.set(bmOBLow, freeSlot, na)
    
    lastOppBar = f_findLastOppositeBar(dir, markerLookback)

    //────────────────────
    // ACTIVE SLOTS — timeout stale; run COLLECT / WAIT_BREAK per slot
    //────────────────────
    for s = 0 to maxBMSeq - 1
        i = base + s
        state = array.get(bmState, i)

        if state == BreakState.IDLE
            continue
        
        left = array.get(bmLeft, i)

        if not na(left)
            if bar_index - left > markerLookback
                array.set(bmState, i, BreakState.IDLE)
                array.set(bmLeft, i, na)
                array.set(bmOBHigh, i, na)
                array.set(bmOBLow, i, na)
                array.set(bmHigh, i, na)
                array.set(bmLow, i, na)
                array.set(bmPreHigh, i, na)
                array.set(bmPreLow, i, na)
                array.set(bmLocked, i, false)
                continue

        //────────────────────
        // COLLECT — stack opposite wicks until zone + left bar set → WAIT_BREAK
        //────────────────────
        if state == BreakState.COLLECT_OPPOSITE

            if f_isOpposite(dir)
                h = array.get(bmHigh, i)
                l = array.get(bmLow, i)


                array.set(bmHigh, i, na(h) ? high : math.max(h, high))
                array.set(bmLow, i, na(l) ? low : math.min(l, low))
                if not na(lastOppBar)

                    array.set(bmLeft, i, lastOppBar)

                    offset = bar_index - lastOppBar

                    refHigh = high[offset]
                    refLow  = low[offset]
                    refOpen = open[offset]

                    [adjTop, adjBot] = f_applyWickRule(refHigh, refLow, refOpen, high[offset - 1], low[offset - 1], dir)

                    array.set(bmOBHigh, i, adjTop)
                    array.set(bmOBLow,  i, adjBot)

                    array.set(bmState, i, BreakState.WAIT_BREAK)

        //────────────────────
        // WAIT_BREAK — touch cancel or impulse breaks ref cluster → Break box
        //────────────────────
        if array.get(bmState, i) == BreakState.WAIT_BREAK and barstate.isconfirmed

            // cancel if touched
            if f_sameDirTouch(dir,array.get(bmOBHigh, i), array.get(bmOBLow, i),array.get(bmLeft, i), false)

                array.set(bmState, i, BreakState.IDLE)
                array.set(bmHigh, i, na)
                array.set(bmLow, i, na)
                array.set(bmLeft, i, na)
                array.set(bmOBHigh, i, na)
                array.set(bmOBLow, i, na)

            // impulse break
            else if f_isImpulse(dir)

                [fbmHigh, fbmLow, fbmPreHigh, fbmPreLow] = f_bmHighLow(dir, markerLookback)

                array.set(bmHigh, i, fbmHigh)
                array.set(bmLow, i, fbmLow)

                // if dir == -1
                array.set(bmPreLow, i, fbmPreLow)
                array.set(bmPreHigh, i, fbmPreHigh)

                highestBearish = array.get(bmHigh, i)
                lowestBullish  = array.get(bmLow, i)

                // if dir == 1
                highestBearish := math.max(highestBearish, array.get(bmPreHigh, i))
                lowestBullish := math.min(lowestBullish, array.get(bmPreLow, i))

                refBreak = dir == 1 ? highestBearish : lowestBullish
                broken = dir == 1 ? high > refBreak : low < refBreak

                if broken and not array.get(bmLocked, i)

                    zoneHigh = array.get(bmOBHigh, i)
                    zoneLow  = array.get(bmOBLow, i)

                    if f_distanceOk(dir == 1 ? zoneHigh : zoneLow, dir == 1 ? lowestBullish : highestBearish) and useBreakMarker and not f_boxLeftExists(array.get(bmLeft, i))
                        bx = box.new( array.get(bmLeft, i), zoneHigh, bar_index, zoneLow, bgcolor = dir == 1 ? bullFill : bearFill, border_color = dir == 1 ? bullBorder : bearBorder)
                        f_boxRegionLabel(array.get(bmLeft, i), bar_index, zoneHigh, "Break marker", dir == 1 ? color.new(color.blue, 78) : color.new(color.red, 78))

                        array.push(obBoxes, bx)
                        array.push(obTops, zoneHigh)
                        array.push(obBots, zoneLow)
                        array.push(obDirs, dir)

                        array.set(bmLocked, i, true)

                        if dir == 1
                            bullConfirmed := true
                        else
                            bearConfirmed := true

                        // reset slot
                        array.set(bmState, i, BreakState.IDLE)
                        array.set(bmLeft, i, na)
                        array.set(bmOBHigh, i, na)
                        array.set(bmOBLow, i, na)

//────────────────────────────────────────
// GENERAL MARKER — rolling gm* extremes (optional use with distance / context)
//────────────────────────────────────────
if isBear
    gmLowestBearishLow := na(gmLowestBearishLow) ? low : math.min(gmLowestBearishLow, low)

if isBull
    gmHighestBullishHigh := na(gmHighestBullishHigh) ? high : math.max(gmHighestBullishHigh, high)

//────────────────────────────────────────
// GENERAL MARKER OB — BULL: BOS → PENDING (last bear zone) → AWAIT_SWEEP → sweep/PB → “Gen” box
//────────────────────────────────────────
if bullBreak and bullState == OBState.IDLE and showBullishOB and not isNeutral
    off = bar_index - lastBearBar
    if off >= 1 and not na(lastBearBar)
        [zTop, zBot] = f_calcBullZone(off)
        bullTop  := zTop
        bullBot  := zBot
        bullLeft := lastBearBar
        bullExtraBearHigh := na
        bullState := OBState.PENDING
        gmLowestBearishLow := na

        if showDebugLabels
            label.new(bar_index, high, "BULL PENDING", style=label.style_label_down, textcolor=color.white)

if bullState == OBState.PENDING
    if f_sameDirTouch(1, bullTop, bullBot, bullLeft, false)
        bullState := OBState.IDLE
        bullTop := na
        bullBot := na
        bullLeft := na

if bullState == OBState.PENDING and barstate.isconfirmed and isBear and not isNeutral
    if f_markerBullLookback()
        bullConfirmHigh := high
        bullConfirmBar  := bar_index
        bullExtraBearHigh := na
        bullImpulseLow := low
        bullState := OBState.AWAIT_SWEEP
        if showDebugLabels
            label.new(bar_index, low, "BULL CONFIRM", style=label.style_label_up, textcolor=color.white)
    else
        bullState := OBState.IDLE

if bullState == OBState.AWAIT_SWEEP
    bullImpulseLow := math.min(bullImpulseLow, low)

    if f_sameDirTouch(1, bullTop, bullBot, bullLeft, false)
        bullState := OBState.IDLE
        bullTop := na
        bullBot := na
        bullLeft := na
        bullConfirmHigh := na
        bullConfirmBar := na
        bullExtraBearHigh := na
        bullImpulseLow := na

    // capture extra bearish high for Sweep2 (first bearish after confirm with lower high than confirm)
    if na(bullExtraBearHigh) and isBear and not na(bullConfirmHigh) and high < bullConfirmHigh
        bullExtraBearHigh := high

    sweep1 = not na(bullConfirmHigh) and high > bullConfirmHigh
    sweep2 = not na(bullExtraBearHigh) and high > bullExtraBearHigh
    pbOk   = not na(bullConfirmBar) and f_pbBull(bullConfirmBar)
    //label.new(bar_index, low, str.tostring(high), style=label.style_label_down, textcolor=color.white)

    if (sweep1 or sweep2 or pbOk) and isBull

        if f_distanceOk(bullTop, bullImpulseLow) and useGeneralMarker and not f_boxLeftExists(bullLeft)
            right = bar_index
            bx = box.new(bullLeft, bullTop, right, bullBot, bgcolor=bullFill, border_color=bullBorder)
            f_boxRegionLabel(bullLeft, right, bullTop, "General OB", color.new(color.blue, 78))
            array.push(obBoxes, bx)
            array.push(obTops, bullTop)
            array.push(obBots, bullBot)
            array.push(obDirs, 1)
            // f_pruneOldest()
            bullConfirmed := true
        bullState := OBState.IDLE

//────────────────────────────────────────
// GENERAL MARKER OB — BEAR: mirror of bull with bear BOS and last bull zone
//────────────────────────────────────────
if bearBreak and bearState == OBState.IDLE and showBearishOB and not isNeutral
    off = bar_index - lastBullBar
    if off >= 1 and not na(lastBullBar)
        [zTop, zBot] = f_calcBearZone(off)
        bearTop  := zTop
        bearBot  := zBot
        bearLeft := lastBullBar
        bearExtraBullLow := na
        bearState := OBState.PENDING
        gmHighestBullishHigh := na

        if showDebugLabels
            label.new(bar_index, low, "BEAR PENDING", style=label.style_label_up, textcolor=color.white)

if bearState == OBState.PENDING
    if f_sameDirTouch(-1, bearTop, bearBot, bearLeft, false)
        bearState := OBState.IDLE
        bearTop := na
        bearBot := na
        bearLeft := na

if bearState == OBState.PENDING and barstate.isconfirmed and isBull  and not isNeutral
    if f_markerBearLookback()
        bearConfirmLow := low
        bearConfirmBar := bar_index
        bearExtraBullLow := na
        bearImpulseHigh := high
        bearState := OBState.AWAIT_SWEEP
        if showDebugLabels
            label.new(bar_index, high, "BEAR CONFIRM", style=label.style_label_down, textcolor=color.white)
    else
        bearState := OBState.IDLE

if bearState == OBState.AWAIT_SWEEP
    bearImpulseHigh := math.max(bearImpulseHigh, high)

    if f_sameDirTouch(-1, bearTop, bearBot, bearLeft, false)
        bearState := OBState.IDLE
        bearTop := na
        bearBot := na
        bearLeft := na
        bearConfirmLow := na
        bearConfirmBar := na
        bearExtraBullLow := na
        bearImpulseHigh := na

    // capture extra bullish low for Sweep2 (first bullish after confirm with higher low than confirm)
    if na(bearExtraBullLow) and isBull and not na(bearConfirmLow) and low > bearConfirmLow
        bearExtraBullLow := low

    sweep1 = not na(bearConfirmLow) and low < bearConfirmLow
    sweep2 = not na(bearExtraBullLow) and low < bearExtraBullLow
    pbOk   = not na(bearConfirmBar) and f_pbBear(bearConfirmBar)
    //label.new(bar_index, high, str.tostring(sweep2), style=label.style_label_down, textcolor=color.white)

    if (sweep1 or sweep2 or pbOk) and isBear
        if f_distanceOk(bearBot, bearImpulseHigh) and useGeneralMarker and not f_boxLeftExists(bearLeft)
            right = bar_index
            bx = box.new(bearLeft, bearTop, right, bearBot, bgcolor=bearFill, border_color=bearBorder)
            f_boxRegionLabel(bearLeft, right, bearTop, "General OB", color.new(color.red, 78))
            array.push(obBoxes, bx)
            array.push(obTops, bearTop)
            array.push(obBots, bearBot)
            array.push(obDirs, -1)
            // f_pruneOldest()
            bearConfirmed := true
        bearState := OBState.IDLE

// if barstate.isconfirmed
//     f_pruneOldest()
//────────────────────────────────────────
// MITIGATION — any touch of stored zone: flag + optional delete box or fade style
//────────────────────────────────────────
sz = array.size(obBoxes)
if sz > 0
    for idx = 0 to sz - 1
        i = (sz - 1) - idx
        bx  = array.get(obBoxes, i)
        top = array.get(obTops, i)
        bot = array.get(obBots, i)
        dir = array.get(obDirs, i)

        // Simple mitigation if it's not same-direction
        touched = math.max(high, low) >= bot and math.min(high, low) <= top
        if touched
            if dir == 1
                bullMitigated := true
            else
                bearMitigated := true

            if deleteMitigatedBoxes
                box.delete(bx)
                array.remove(obBoxes, i)
                array.remove(obTops, i)
                array.remove(obBots, i)
                array.remove(obDirs, i)
            else if fadeMitigated
                box.set_bgcolor(bx, color.new(dir == 1 ? bullFill : bearFill, 85))
                box.set_border_color(bx, mitBorder)

//────────────────────────────────────────
// HTF ZONE TOUCH — request.security HTF bar range vs obTops/obBots; fire once per HTF close
//────────────────────────────────────────
ztfSecHigh = request.security(syminfo.tickerid, zoneTouchTimeframe, high, barmerge.gaps_off, barmerge.lookahead_off)
ztfSecLow = request.security(syminfo.tickerid, zoneTouchTimeframe, low, barmerge.gaps_off, barmerge.lookahead_off)
ztfNewBar = timeframe.change(zoneTouchTimeframe)
ztfRh = ztfSecHigh[1]
ztfRl = ztfSecLow[1]
ztfRhEff = alertOnZoneTouchHTF and ztfNewBar and barstate.isconfirmed ? ztfRh : na
ztfRlEff = alertOnZoneTouchHTF and ztfNewBar and barstate.isconfirmed ? ztfRl : na
[ztfTouchBull, ztfTouchBear] = f_zonesTouchedInRange(obTops, obBots, obDirs, ztfRhEff, ztfRlEff)
ztfTouchAny = ztfTouchBull or ztfTouchBear
htfZoneTouchAlert = alertOnZoneTouchHTF and ztfNewBar and barstate.isconfirmed and ztfTouchAny

if htfZoneTouchAlert
    alert("OB zone touched on " + str.tostring(zoneTouchTimeframe) + " candle (" + (ztfTouchBull and ztfTouchBear ? "bull+bear" : ztfTouchBull ? "bullish" : "bearish") + ")", alert.freq_once_per_bar_close)
